CREATE
VIEW VIEW_CONCILIA_FORNECEDOR
AS

select
	PP.ID,
	PP.ID_FIN_PARCELA_PAGAR,
	PP.ID_FIN_TIPO_PAGAMENTO, 
	PP.ID_CONTA_CAIXA,
	PP.DATA_PAGAMENTO,
	PP.TAXA_JURO,
	PP.TAXA_MULTA,
	PP.TAXA_DESCONTO,
	PP.VALOR_JURO,
	PP.VALOR_MULTA,
	PP.VALOR_DESCONTO,
	PP.VALOR_PAGO,
	PP.HISTORICO,
	LP.ID_FORNECEDOR
from
	FIN_LANCAMENTO_PAGAR LP
	INNER JOIN FIN_PARCELA_PAGAR PPR ON (PPR.ID_FIN_LANCAMENTO_PAGAR = LP.ID)
	INNER JOIN FIN_PARCELA_PAGAMENTO PP ON (PP.ID_FIN_PARCELA_PAGAR = PPR.ID)
